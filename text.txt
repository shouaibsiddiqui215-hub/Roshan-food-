<script>
const selectAllBtn = document.getElementById('selectAllBtn');
let allSelected = false;
selectAllBtn.addEventListener('click', () => {
    const checkboxes = document.querySelectorAll('input[name="product"]');
    allSelected = !allSelected;
    checkboxes.forEach(cb => cb.checked = allSelected);
});

document.getElementById('orderForm').addEventListener('submit', function(e){
    e.preventDefault();
    const name = this.name.value;
    const phone = this.phone.value;
    const address = this.address.value;
    const products = [];
    const checkboxes = this.querySelectorAll('input[name="product"]');
    checkboxes.forEach(cb=>{
        if(cb.checked){
            const qtyName='quantity_'+cb.value.replace(/\s+/g,'_').replace(/₹/g,'');
            const qty = this[qtyName].value;
            products.push(`${cb.value} x ${qty}`);
        }
    });
    if(products.length===0){
        alert('Please select at least one product!');
        return;
    }
    alert(`Thank you ${name}! Your order has been received.\n\nProducts:\n${products.join('\n')}\n\nWe will contact you at ${phone}.`);
    this.reset();
});
</script>
